# Command for API
gradle executeMedeisv2e4ApccApiFeature -DcucumberTags='@regression'

# Windows PowerShell
gradle executeMedeisv2e4ApccApiFeature -DenvironmentName=QA-LOCAL -DcucumberTags="@regression"
# Windows Command line
gradle.bat executeMedeisv2e4ApccApiFeature -DenvironmentName=QA-LOCAL -DcucumberTags="@regression"

# Command for UI
gradle executeMedeisv2e4ApccWebFeature -Dbrowser=CHROME_DOCKER -DcucumberTags='@regression'

gradle executeMedeisv2e4ApccWebFeature -Dbrowser=CHROME -DcucumberTags='@regression'

# Windows Command line
gradle.bat executeMedeisv2e4ApccWebFeature -DenvironmentName=QA-LOCAL -Dbrowser=CHROME -DcucumberTags="@regression"

# Windows Command line with headless mode
gradle.bat executeMedeisv2e4ApccWebFeature -DenvironmentName=QA-LOCAL -Dbrowser=CHROME -DheadlessMode=true -DcucumberTags="@regression"
